接口定义： http://hostname:port/:endppint


1. 开始活动: start
Method: GET
request:{
  start: 1
}
response:{
  data:{
    flag: 1
  }
}

2. 人员名单： staff
Method: GET
request:{}
response:{
  "data": [
    {"name":"@FIRST", id":1 }
    ...
  ]
}

3. 奖池设置：lottery
Method: GET
request:{}
response:{
  "data":[
    {"level": "一等奖", "id":1, "limit":2,  "info": "@WORD"}
    ...
  ]
}

4. 中奖人：lucky
Method: GET
request:{
  "grade":id
}
response:{
  data:{
    id: 1
  }
}

5. 取消中奖人：luckless
Method: GET
request{
  id: 1
}
response:{
  data:{
    flag: 1
  }
}

6. 中奖名单： luckylist
Method: GET
request:{}
response:{
  data:[
    {"level":"一等奖"， list:["name","name"]}
  ]
}
